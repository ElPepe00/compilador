// 1. Declaració de la funció 'sumar'
funcio int sumar(int a, int b):
    int res;
    res = a + b;
    return res;
funcio_fi

    // 2. Ús de la funció
    int x = 15;
    int y = 10;
    
    // El resultat es guarda en una variable local
    int total = sumar(x, y);
    
    // 3. Estructura condicional emprant el resultat
    if (total > 20):
        // Si és major que 20, imprimim true
        imprimir(true);
    else:
        // Si no, imprimim false
        imprimir(false);
    else_fi
